# Shopify App Environment Variables
# Bu dosyayı .env olarak kaydedin ve değerleri doldurun

# 1. SHOPIFY_API_KEY
# Shopify Partners Dashboard'dan alın: https://partners.shopify.com
# Mevcut değer (shopify.app.toml'den): 39e3fc3eda0c39b0097c956dd53f0e3e
SHOPIFY_API_KEY=39e3fc3eda0c39b0097c956dd53f0e3e

# 2. SHOPIFY_API_SECRET
# Shopify Partners Dashboard > App Settings > Client credentials bölümünden alın
SHOPIFY_API_SECRET=your_shopify_api_secret_here

# 3. SCOPES
# Ürün ve envanter yönetimi için gerekli izinler
SCOPES=read_products,write_products,read_inventory,write_inventory

# 4. APP_URL
# Production: https://your-app-domain.com
# Development: Shopify CLI otomatik ayarlar
APP_URL=https://example.com

# 5. HOST
# Production: https://your-app-domain.com
# Development: localhost veya ngrok/cloudflare tunnel URL'i
HOST=https://example.com

# 6. DATABASE_URL
# SQLite (default): sqlite:./database.sqlite
# PostgreSQL: postgresql://user:password@localhost:5432/dbname
# MySQL: mysql://user:password@localhost:3306/dbname
DATABASE_URL=sqlite:./database.sqlite

# 7. OPENAI_API_KEY
# OpenAI API anahtarı: https://platform.openai.com/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# 8. ENCRYPTION_KEY
# 32 karakterlik random string (session şifreleme için)
# Oluşturmak için: openssl rand -hex 32
ENCRYPTION_KEY=your_32_character_encryption_key_here
